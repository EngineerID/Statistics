SQL Syntax Reference

SQL Command Syntax Example Description
AND / OR SELECT col
FROM table
WHERE cond1
AND|OR cond2;		 SELECT * FROM employees
WHERE department = 'Sales'
AND salary > 50000;		 Used to filter records based on more than one condition.
ANY / ALL SELECT col
FROM table
WHERE col operator ANY|ALL
(subquery);		 SELECT name FROM products
WHERE price > ANY
(SELECT price FROM orders WHERE quantity > 10);		 "ANY" returns true if any subquery values meet the condition; "ALL" requires all to meet it.
ALTER TABLE ALTER TABLE table
ADD|DROP|ALTER COLUMN col datatype;		 ALTER TABLE customers
ADD email_address VARCHAR(255);		 Adds, deletes, or modifies columns in an existing table.
AS (Aliases) SELECT col AS alias
FROM table;		 SELECT employee_name AS "Staff Name"
FROM employees;		 Gives columns or tables a temporary name for readability.
AVG() SELECT AVG(col)
FROM table;		 SELECT AVG(unit_price)
FROM inventory;		 Returns the average value of a numeric column.
BETWEEN WHERE col
BETWEEN v1 AND v2;		 SELECT * FROM orders
WHERE order_date
BETWEEN '2024-01-01' AND '2024-03-31';		 Filters values within a given range (inclusive).
CASE CASE
  WHEN cond1 THEN res1
  ELSE resN
END;		 SELECT product_name,
CASE WHEN stock < 10 THEN 'Low Stock'
ELSE 'In Stock'
END AS "Status"
FROM warehouse;		 Handles conditional logic (IF-THEN-ELSE).
CONCAT CONCAT(s1, s2, ...); SELECT CONCAT(first_name, ' ', last_name) AS "Full Name"
FROM employees;		 Joins multiple strings into one.
COUNT SELECT COUNT(col) FROM table; SELECT COUNT(customer_id) FROM orders; Returns the number of rows matching the criteria.
CREATE TABLE CREATE TABLE table (
  col1 data_type,
  col2 data_type
);		 CREATE TABLE inventory (
  item_id INT,
  item_name VARCHAR(100)
);		 Creates a new table in the database.
DELETE DELETE FROM table
WHERE condition;		 DELETE FROM customers
WHERE last_login < '2020-01-01';		 Deletes specific rows. Caution: Omitting WHERE deletes all data.
EXTRACT SELECT EXTRACT(part FROM col); SELECT EXTRACT(month FROM hire_date) AS "Hire Month"
FROM employees;		 Extracts a portion of a date (Year, Month, Day, etc.).
GROUP BY SELECT col, aggregate(col)
FROM table
GROUP BY col;		 SELECT region, SUM(sales_total)
FROM store_data GROUP BY region;		 Groups rows sharing the same values into summary rows.
HAVING SELECT col FROM table
GROUP BY col
HAVING condition;		 SELECT category, COUNT(*)
FROM products GROUP BY category
HAVING COUNT(*) > 20;		 Filters groups (used instead of WHERE for aggregates).
JOINS SELECT * FROM T1
JOIN T2 ON T1.id = T2.id;		 SELECT orders.id, customers.name
FROM orders
INNER JOIN customers ON orders.cust_id = customers.id;		 Combines rows from tables based on a related column.
LIKE / ILIKE WHERE col LIKE 'pattern'; WHERE email LIKE '%@gmail.com'; Pattern matching. % is a wildcard. ILIKE is case-insensitive.
MAX() / MIN() SELECT MAX|MIN(col) FROM table; SELECT MAX(salary) FROM employees; Returns the largest or smallest value.
SELECT TOP / LIMIT SELECT TOP (n) | LIMIT n; SELECT * FROM sales_logs
ORDER BY date DESC LIMIT 10;		 Restricts the number of rows returned.
UPDATE UPDATE table
SET col = val
WHERE condition;		 UPDATE products SET price = 19.99
WHERE sku = 'A100';		 Modifies existing records in a table.
Math (+, -, *, /) SELECT col1 * col2; SELECT (price * quantity) AS "Total Revenue"
FROM order_items;		 Performs calculations on numeric data.
