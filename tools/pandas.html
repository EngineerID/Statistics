Importing and Creating Data

Keyword Description Syntax Example
Import Pandas Imports the Pandas library for data manipulation. import pandas as pd import pandas as pd # Pandas successfully imported
DataFrame Creates a two-dimensional tabular data structure. pd.DataFrame(data, columns=[...]) import pandas as pd
data_dict = {'City': ['New York', 'Los Angeles', 'Chicago'], 'Population': [8419000, 3893000, 2716000]}
df = pd.DataFrame(data_dict)
print(df)
Series Creates a one-dimensional labeled array. pd.Series(data) import pandas as pd
cities = ['New York', 'Los Angeles', 'Chicago']
series = pd.Series(cities)
print(series)
read_csv() Reads a comma-separated values (CSV) file into a DataFrame. pd.read_csv(filepath_or_buffer) import pandas as pd
# Assuming 'cities.csv' contains city data
df = pd.read_csv('cities.csv')
print(df.head())
read_excel() Reads an Excel file into a DataFrame. pd.read_excel(io, sheet_name=0) import pandas as pd
# Assuming 'cities.xlsx' contains city data
df = pd.read_excel('cities.xlsx')
print(df.head())

Exporting Data

Keyword Description Syntax Example
to_csv() Saves your DataFrame to a CSV file. Use index=False to avoid saving the row numbers as a new column. df.to_csv('filename.csv', index=False) import pandas as pd

Selecting and Indexing Data

Keyword Description Syntax Example
loc[] Accesses a group of rows and columns by labels. df.loc[labels] import pandas as pd
df = pd.DataFrame({'Population': [8419000, 3893000]}, index=['New York', 'Los Angeles'])
print(df.loc['New York'])
iloc[] Accesses a group of rows and columns by integer positions. df.iloc[positions] import pandas as pd
df = pd.DataFrame({'City': ['New York', 'Los Angeles']})

set_index() Sets the DataFrame index using existing columns. df.set_index(keys, inplace=False) import pandas as pd

reset_index() Resets the index to the default integer index. df.reset_index(inplace=False) import pandas as pd
df = pd.DataFrame({'Population': [8419000, 3893000]}, index=['New York', 'Los Angeles'])
df.reset_index(inplace=True)
print(df)

Filtering Data

Keyword Description Syntax Example
Boolean Operators Combines multiple conditions for filtering. (condition1) & (condition2), |, ~ import pandas as pd
df = pd.DataFrame({'Population': [8419000, 3893000, 2716000]})
filtered = df[(df['Population'] > 3000000) & (df['Population'] < 8000000)]
print(filtered)
isin() Checks whether each element is contained in values. series.isin(values) import pandas as pd
df = pd.DataFrame({'City': ['New York', 'Los Angeles', 'Chicago']})
filtered = df[df['City'].isin(['New York', 'Chicago'])]
print(filtered)
query() Queries the DataFrame with a boolean expression. df.query(expr) import pandas as pd
df = pd.DataFrame({'Population': [8419000, 3893000]})
filtered = df.query('Population > 4000000')
print(filtered)

Handling Missing Data

Keyword Description Syntax Example
isna() Detects missing values. df.isna() import pandas as pd
import numpy as np
df = pd.DataFrame({'A': [1, np.nan, 3]})
print(df.isna())

print(df)
fillna() Fills missing values with a specified value. df.fillna(value, inplace=False) import pandas as pd
import numpy as np
df = pd.DataFrame({'A': [1, np.nan, 3]})

print(df)

Data Cleaning

Keyword Description Syntax Example
unique() / nunique() unique() returns an array of all unique values; nunique() returns the count of unique values. df['column'].unique() / df['column'].nunique() import pandas as pd
df = pd.DataFrame({'City': ['New York', 'Los Angeles', 'New York']})
print(df['City'].unique())
print(df['City'].nunique()) # 2
drop_duplicates() Removes rows that are exactly the same across all columns. df.drop_duplicates(inplace=True) import pandas as pd
df = pd.DataFrame({'City': ['New York', 'New York'], 'Population': [8419000, 8419000]})
df.drop_duplicates(subset=['City'], keep='first', inplace=True)
print(df)

String Operations

Keyword Description Syntax Example
str.contains() Filters rows based on whether a string contains a specific substring. df[df['col'].str.contains('text')] import pandas as pd
df = pd.DataFrame({'City': ['New York', 'Los Angeles', 'Chicago']})
big_cities = df[df['City'].str.contains('New')]
print(big_cities)
str.upper() / str.lower() Mass-transforms the casing of an entire column of text. df['col'].str.upper() import pandas as pd
df = pd.DataFrame({'City': ['new york', 'los angeles']})
df['City'] = df['City'].str.upper()
print(df)

Applying Functions

Keyword Description Syntax Example
apply() Applies a function along an axis of the DataFrame. df.apply(func, axis=0) import pandas as pd
df = pd.DataFrame({'A': [1, 2], 'B': [3, 4]})
print(df.apply(lambda x: x.max() - x.min()))
map() Maps values of Series according to input correspondence. series.map(arg) import pandas as pd
s = pd.Series(['cat', 'dog', 'bird'])
print(s.map({'cat': 'kitten', 'dog': 'puppy'}))
applymap() Applies a function to every element of the DataFrame. df.applymap(func) import pandas as pd
df = pd.DataFrame([[1, 2], [3, 4]])
print(df.applymap(lambda x: x**2))

Grouping and Aggregating

Keyword Description Syntax Example
groupby() Groups DataFrame using a mapper or by columns. df.groupby(by) import pandas as pd
df = pd.DataFrame({'Animal': ['Dog', 'Dog', 'Cat'], 'Weight': [25, 30, 10]})
grouped = df.groupby('Animal')
print(grouped.mean())
agg() Aggregates using one or more operations. grouped.agg(func) import pandas as pd
df = pd.DataFrame({'Animal': ['Dog', 'Dog'], 'Weight': [25, 30]})
print(df.groupby('Animal').agg('mean'))
value_counts() Returns counts of unique values. series.value_counts() import pandas as pd
df = pd.DataFrame({'Animal': ['Dog', 'Cat', 'Dog']})
print(df['Animal'].value_counts())
pivot_table() Creates a spreadsheet-style pivot table. pd.pivot_table(df, values, index, columns) import pandas as pd
df = pd.DataFrame({'A': ['foo', 'bar'], 'B': ['one', 'one'], 'C': [1, 2]})
print(pd.pivot_table(df, values='C', index='A', columns='B', aggfunc='sum'))

Merging and Concatenating

Keyword Description Syntax Example
concat() Concatenates pandas objects along an axis. pd.concat(objs, axis=0) import pandas as pd
df1 = pd.DataFrame({'A': [1, 2]})
df2 = pd.DataFrame({'A': [3, 4]})
print(pd.concat([df1, df2]))
merge() Merges DataFrames using database-style join. pd.merge(left, right, how='inner', on=None) import pandas as pd
left = pd.DataFrame({'key': ['K0', 'K1'], 'A': ['A0', 'A1']})
right = pd.DataFrame({'key': ['K0', 'K1'], 'B': ['B0', 'B1']})
print(pd.merge(left, right, on='key'))
join() Joins columns with other DataFrame on index or column. left.join(right, how='left') import pandas as pd
left = pd.DataFrame({'A': ['A0', 'A1']}, index=['K0', 'K1'])
right = pd.DataFrame({'B': ['B0', 'B1']}, index=['K0', 'K1'])
print(left.join(right))

Reshaping Data

Keyword Description Syntax Example
melt() "Unpivots" a DataFrame from wide format to long format (useful for visualization libraries). pd.melt(df, id_vars=['col']) import pandas as pd
df = pd.DataFrame({'City': ['New York', 'Los Angeles'], 'Population': [8419000, 3893000]})
long_df = pd.melt(df, id_vars=['City'], value_vars=['Population'])
print(long_df)
pivot() Reshapes data from long to wide format. df.pivot(index, columns, values) import pandas as pd
df = pd.DataFrame({'date': ['2023-01-01', '2023-01-02'], 'city': ['New York', 'New York'], 'temp': [32, 34]})
pivoted = df.pivot(index='date', columns='city', values='temp')
print(pivoted)

Time Series Operations

Keyword Description Syntax Example
resample() Resamples time-series data to a different frequency (e.g., daily to monthly). df.resample(rule).agg(func) import pandas as pd
import numpy as np
rng = pd.date_range('2023-01-01', periods=10, freq='D')
df = pd.DataFrame({'value': np.arange(10)}, index=rng)
monthly = df.resample('M').mean()
print(monthly)
rolling() Provides rolling window calculations, like moving averages. df.rolling(window).mean() import pandas as pd
df = pd.DataFrame({'value': [1, 2, 3, 4, 5]})
rolling_mean = df.rolling(window=3).mean()
print(rolling_mean)
shift() Shifts index by desired number of periods, useful for time series lag features. df.shift(periods=1) import pandas as pd
df = pd.DataFrame({'value': [1, 2, 3]})
df['lagged'] = df['value'].shift(1)
print(df)
asfreq() Converts time series to specified frequency, handling missing data. df.asfreq(freq) import pandas as pd
import numpy as np
rng = pd.date_range('2023-01-01', periods=3, freq='D')
df = pd.DataFrame({'value': [1, 2, 3]}, index=rng)
hourly = df.asfreq('H')
print(hourly)
